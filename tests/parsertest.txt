
var n = 100;
func fib(n) {
    if(n <= 2) {
        return 1;
    }
    return fib(n-1) + fib(n - 2);
}

func fact(n) {
    if(n == 1) {
        return n;
    }
    return n * fact(n-1);
}

func add(x, y) {
    return x + y;
}
