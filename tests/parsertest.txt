function fact(n) {
    if(n == 1) {
        return 1;
    }
    return n * fact(n - 1);
}

function add( a, b ) {
    return a + b; 
}


function fibo(n) {
    if(n < 2) {
        return n;
    }
    return (fibo(n - 1) + fibo(n - 2));
}
